  var a = $(this).bounds(),
      b = element.bounds();

  return(
    (a.left >= b.left) &&
    (a.right <= b.right) &&
    (a.top >= b.top) &&
    (a.bottom <= b.bottom)

jQuery.fn.toOverlap = function(element) {
  var a = $(this).bounds(),
      b = element.bounds();

  return( !(
      (a.bottom < b.top) ||
      (a.top > b.bottom) ||
      (a.right < b.left) ||
      (a.left > b.right)
    )
  );
};
